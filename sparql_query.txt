PREFIX foaf: <http://www.owl-ontologies.com/Ontology1392222449.owl#>
SELECT ?name ?x ?y (SUM(?val) AS ?summedval)
 WHERE { ?x foaf:personHasName  ?name; foaf:hasAccount ?y .?y foaf:credit ?val
         FILTER regex(?name, "a", "i")   }
